# ==================== LOGGING ====================
def setup_logger():
    """Configure structured logging"""
    logger = logging.getLogger("KynxSS")
    logger.setLevel(getattr(logging, Config.LOG_LEVEL))
    
    # File handler with rotation
    log_file = Config.LOGS_DIR / f"kynx_{datetime.now().strftime('%Y%m%d')}.log"
    file_handler = logging.FileHandler(log_file, encoding='utf-8')
    file_handler.setFormatter(logging.Formatter(
        '%(asctime)s | %(levelname)-8s | %(name)s | %(message)s'
    ))
    
    # Console handler
    console_handler = logging.StreamHandler()
    console_handler.setFormatter(logging.Formatter(
        '%(asctime)s | %(levelname)-8s | %(message)s'
    ))
    
    logger.addHandler(file_handler)
    logger.addHandler(console_handler)
    return logger

# ==================== CACHE MANAGER ====================
class CacheManager:
    """Advanced caching system"""
    
    def __init__(self, ttl: int = Config.CACHE_TTL):
        self.cache: Dict[str, tuple[Any, float]] = {}
        self.ttl = ttl
        self._cleanup_task = None
    
    def start_cleanup(self):
        """Start periodic cache cleanup"""
        async def cleanup():
            while True:
                await asyncio.sleep(300)  # 5 minutes
                self._cleanup()
        
        self._cleanup_task = asyncio.create_task(cleanup())
    
    def _cleanup(self):
        """Remove expired entries"""
        now = time.time()
        expired = [k for k, (_, ts) in self.cache.items() if now - ts > self.ttl]
        for k in expired:
            del self.cache[k]
        if expired:
            logger.debug(f"Cleaned {len(expired)} cache entries")
    
    def get(self, key: str) -> Optional[Any]:
        """Get value from cache"""
        if key in self.cache:
            value, timestamp = self.cache[key]
            if time.time() - timestamp <= self.ttl:
                return value
            else:
                del self.cache[key]
        return None
    
    def set(self, key: str, value: Any):
        """Set value in cache"""
        self.cache[key] = (value, time.time())
    
    def invalidate(self, key: str):
        """Invalidate cache entry"""
        self.cache.pop(key, None)
    
    def invalidate_pattern(self, pattern: str):
        """Invalidate all keys matching pattern"""
        to_delete = [k for k in self.cache if pattern in k]
        for k in to_delete:
            del self.cache[k]

# ==================== API CLIENT ====================
class GameServerAPI:
    """Async API client for game server communication"""
    
    def __init__(self, base_url: str):
        self.base_url = base_url
        self.session: Optional[aiohttp.ClientSession] = None
        self.cache = CacheManager()
        self.retry_count = 3
        self.retry_delay = 1
    
    async def ensure_session(self):
        """Ensure aiohttp session exists"""
        if self.session is None or self.session.closed:
            self.session = aiohttp.ClientSession(
                headers={"User-Agent": "KynxSS/10.0"},
                timeout=aiohttp.ClientTimeout(total=10)
            )
    
    async def close(self):
        """Close session"""
        if self.session and not self.session.closed:
            await self.session.close()
    
    async def _request(self, method: str, endpoint: str, **kwargs) -> Optional[Dict]:
        """Make HTTP request with retries"""
        await self.ensure_session()
        url = f"{self.base_url}/{endpoint.lstrip('/')}"
        
        for attempt in range(self.retry_count):
            try:
                async with self.session.request(method, url, **kwargs) as response:
                    if response.status == 200:
                        return await response.json()
                    elif response.status == 404:
                        return None
                    else:
                        logger.warning(f"API error {response.status}: {await response.text()}")
            except asyncio.TimeoutError:
                logger.warning(f"Timeout on attempt {attempt + 1}/{self.retry_count}")
            except Exception as e:
                logger.error(f"Request error: {e}")
            
            if attempt < self.retry_count - 1:
                await asyncio.sleep(self.retry_delay * (attempt + 1))
        
        return None
    
    async def get_players(self) -> Dict[str, Any]:
        """Get online players"""
        cache_key = "players"
        cached = self.cache.get(cache_key)
        if cached:
            return cached
        
        data = await self._request("GET", "get_players")
        if data:
            self.cache.set(cache_key, data)
        return data or {"count": 0, "players": []}
    
    async def get_bans(self) -> List[Dict]:
        """Get all bans"""
        cache_key = "bans"
        cached = self.cache.get(cache_key)
        if cached:
            return cached
        
        data = await self._request("GET", "get_bans")
        if data:
            self.cache.set(cache_key, data)
        return data or []
    
    async def send_command(self, command: str, **kwargs) -> bool:
        """Send command to game server"""
        payload = {
            "command": command,
            "timestamp": time.time(),
            **kwargs
        }
        
        result = await self._request("POST", "send_command", json=payload)
        if result:
            # Invalidate relevant caches
            if command in ["ban", "unban", "pcban", "unpcban"]:
                self.cache.invalidate("bans")
            elif command in ["mute", "umute"]:
                self.cache.invalidate("mutes")
            elif command in ["kick"]:
                self.cache.invalidate("players")
            
            return True
        return False
    
    async def get_stats(self) -> ServerStats:
        """Get server statistics"""
        cache_key = "stats"
        cached = self.cache.get(cache_key)
        if cached:
            return cached
        
        players = await self.get_players()
        bans = await self.get_bans()
        
        stats = ServerStats(
            players_online=players.get("count", 0),
            total_bans=len(bans),
            normal_bans=sum(1 for b in bans if b.get("banType") == "normal"),
            pc_bans=sum(1 for b in bans if b.get("banType") == "pc")
        )
        
        self.cache.set(cache_key, stats)
        return stats

# ==================== ROBLOX API CLIENT ====================
class RobloxAPI:
    """Async Roblox API client"""
    
    def __init__(self):
        self.session: Optional[aiohttp.ClientSession] = None
        self.cache = CacheManager(ttl=3600)  # 1 hour cache
    
    async def ensure_session(self):
        if self.session is None or self.session.closed:
            self.session = aiohttp.ClientSession(
                headers={"User-Agent": "KynxSS/10.0"}
            )
    
    async def close(self):
        if self.session and not self.session.closed:
            await self.session.close()
    
    async def get_user(self, user_id: str) -> Optional[RobloxUser]:
        """Get Roblox user data"""
        cache_key = f"user_{user_id}"
        cached = self.cache.get(cache_key)
        if cached:
            return cached
        
        await self.ensure_session()
        
        try:
            # Get user info
            async with self.session.get(f"https://users.roblox.com/v1/users/{user_id}") as resp:
                if resp.status != 200:
                    return None
                user_data = await resp.json()
            
            # Get avatar
            async with self.session.get(
                f"https://thumbnails.roblox.com/v1/users/avatar-headshot",
                params={"userIds": user_id, "size": "420x420", "format": "Png"}
            ) as resp:
                avatar_data = await resp.json() if resp.status == 200 else {}
            
            user = RobloxUser(
                id=user_id,
                username=user_data.get("name", "Unknown"),
                display_name=user_data.get("displayName", "Unknown"),
                avatar_url=avatar_data.get("data", [{}])[0].get("imageUrl") if avatar_data else None,
                created_at=user_data.get("created")
            )
            
            self.cache.set(cache_key, user)
            return user
            
        except Exception as e:
            logger.error(f"Roblox API error: {e}")
            return None

# ==================== DATABASE MANAGER ====================
class DatabaseManager:
    """Simple JSON-based database"""
    
    def __init__(self, db_path: Path):
        self.db_path = db_path
        self.data: Dict[str, Any] = {}
        self.lock = asyncio.Lock()
        self._load()
    
    def _load(self):
        """Load data from disk"""
        if self.db_path.exists():
            try:
                with open(self.db_path, 'r', encoding='utf-8') as f:
                    self.data = json.load(f)
            except:
                self.data = {}
        
        # Initialize collections
        self.data.setdefault("mod_logs", [])
        self.data.setdefault("notes", {})
        self.data.setdefault("mutes", {})
        self.data.setdefault("settings", {})
        self.data.setdefault("blacklist", [])
    
    async def save(self):
        """Save data to disk"""
        async with self.lock:
            temp_path = self.db_path.with_suffix('.tmp')
            with open(temp_path, 'w', encoding='utf-8') as f:
                json.dump(self.data, f, indent=2)
            temp_path.replace(self.db_path)
    
    async def add_log(self, log: Dict):
        """Add moderation log"""
        self.data["mod_logs"].append({
            **log,
            "log_id": hashlib.md5(f"{time.time()}{log}".encode()).hexdigest()[:8]
        })
        
        # Keep only last 1000 logs
        if len(self.data["mod_logs"]) > 1000:
            self.data["mod_logs"] = self.data["mod_logs"][-1000:]
        
        await self.save()
    
    async def get_user_logs(self, user_id: str, limit: int = 10) -> List[Dict]:
        """Get logs for specific user"""
        return [
            log for log in self.data["mod_logs"][-limit:]
            if log.get("target_id") == user_id
        ][-limit:]
    
    async def add_note(self, user_id: str, note: str, moderator: str):
        """Add note to user"""
        self.data["notes"].setdefault(user_id, []).append({
            "note": note,
            "moderator": moderator,
            "timestamp": time.time()
        })
        await self.save()
    
    async def get_notes(self, user_id: str) -> List[Dict]:
        """Get user notes"""
        return self.data["notes"].get(user_id, [])
    
    async def clear_notes(self, user_id: str):
        """Clear all notes for user"""
        self.data["notes"].pop(user_id, None)
        await self.save()

# ==================== EMBED FACTORY ====================
class EmbedFactory:
    """Factory for creating consistent embeds"""
    
    COLORS = {
        "info": discord.Color.blue(),
        "success": discord.Color.green(),
        "warning": discord.Color.gold(),
        "error": discord.Color.red(),
        "ban": discord.Color.dark_red(),
        "mute": discord.Color.orange(),
        "kick": discord.Color.purple()
    }
    
    ICONS = {
        "info": "ℹ️",
        "success": "✅",
        "warning": "⚠️",
        "error": "❌",
        "ban": "🔨",
        "pcban": "💻",
        "mute": "🔇",
        "kick": "👢",
        "players": "👥",
        "stats": "📊",
        "settings": "⚙️"
    }
    
    @classmethod
    def create(cls, 
               title: str,
               description: str = "",
               color_type: str = "info",
               user: Optional[Union[discord.User, discord.Member]] = None,
               target: Optional[RobloxUser] = None) -> discord.Embed:
        
        icon = cls.ICONS.get(color_type, "")
        embed = discord.Embed(
            title=f"{icon} {title}",
            description=description,
            color=cls.COLORS.get(color_type, discord.Color.blue()),
            timestamp=datetime.utcnow()
        )
        
        if user:
            embed.set_footer(
                text=f"Requested by {user.display_name}",
                icon_url=user.display_avatar.url
            )
        
        if target:
            if target.avatar_url:
                embed.set_thumbnail(url=target.avatar_url)
            embed.add_field(
                name="Target",
                value=f"{target.username} (@{target.display_name})\nID: `{target.id}`",
                inline=False
            )
        
        return embed
    
    @classmethod
    def error(cls, message: str, user: Optional[discord.User] = None) -> discord.Embed:
        """Create error embed"""
        return cls.create("Error", message, "error", user)

class ConfirmView(View):
    """Generic confirmation view"""
    
    def __init__(self, 
                 user: discord.User,
                 callback,
                 cancel_callback=None,
                 timeout: int = 60):
        super().__init__(timeout=timeout)
        self.user = user
        self.callback = callback
        self.cancel_callback = cancel_callback
    
    async def interaction_check(self, interaction: discord.Interaction) -> bool:
        if interaction.user != self.user:
            await interaction.response.send_message(
                "This menu belongs to someone else!", 
                ephemeral=True
            )
            return False
        return True
    
    @discord.ui.button(label="Confirm", style=discord.ButtonStyle.success, emoji="✅")
    async def confirm(self, interaction: discord.Interaction, button: Button):
        await self.callback(interaction)
    
    @discord.ui.button(label="Cancel", style=discord.ButtonStyle.secondary, emoji="❌")
    async def cancel(self, interaction: discord.Interaction, button: Button):
        if self.cancel_callback:
            await self.cancel_callback(interaction)
        else:
            await interaction.response.edit_message(
                content="Action cancelled.",
                embed=None,
                view=None
            )

class PaginatedView(View):
    """Paginated list view"""
    
    def __init__(self, 
                 user: discord.User,
                 items: List[Any],
                 title: str,
                 formatter,
                 items_per_page: int = 5):
        super().__init__(timeout=120)
        self.user = user
        self.items = items
        self.title = title
        self.formatter = formatter
        self.items_per_page = items_per_page
        self.current_page = 0
        self.max_page = max(0, (len(items) - 1) // items_per_page)
        
        self._update_buttons()
    
    def _update_buttons(self):
        """Update button states"""
        self.prev_page.disabled = self.current_page == 0
        self.next_page.disabled = self.current_page >= self.max_page
    
    def _get_page_embed(self) -> discord.Embed:
        """Get embed for current page"""
        start = self.current_page * self.items_per_page
        end = min(start + self.items_per_page, len(self.items))
        
        embed = EmbedFactory.create(
            self.title,
            f"Page {self.current_page + 1}/{self.max_page + 1} • Total: {len(self.items)}",
            "info",
            self.user
        )
        
        for i, item in enumerate(self.items[start:end], start + 1):
            embed.add_field(
                name=f"#{i}",
                value=self.formatter(item),
                inline=False
            )
        
        return embed
    
    async def interaction_check(self, interaction: discord.Interaction) -> bool:
        if interaction.user != self.user:
            await interaction.response.send_message(
                "This menu belongs to someone else!",
                ephemeral=True
            )
            return False
        return True
    
    @discord.ui.button(label="◀", style=discord.ButtonStyle.secondary)
    async def prev_page(self, interaction: discord.Interaction, button: Button):
        self.current_page -= 1
        self._update_buttons()
        await interaction.response.edit_message(embed=self._get_page_embed(), view=self)
    
    @discord.ui.button(label="▶", style=discord.ButtonStyle.secondary)
    async def next_page(self, interaction: discord.Interaction, button: Button):
        self.current_page += 1
        self._update_buttons()
        await interaction.response.edit_message(embed=self._get_page_embed(), view=self)
    
    @discord.ui.button(label="Close", style=discord.ButtonStyle.danger, emoji="✖️")
    async def close(self, interaction: discord.Interaction, button: Button):
        await interaction.response.edit_message(content="Closed.", embed=None, view=None)

class KynxBot(commands.Bot):
    """Main bot class with enhanced functionality"""
    
    def __init__(self):
        intents = discord.Intents.default()
        intents.message_content = False
        intents.members = True
        
        super().__init__(
            command_prefix="!",
            intents=intents,
            help_command=None
        )
        
        # Initialize components

        self.roblox = RobloxAPI()
        self.db = DatabaseManager(Config.BASE_DIR / "database.json")
        self.start_time = time.time()
        
        # Command groups
        self.mod_group = app_commands.Group(name="mod", description="Moderation commands")
        self.admin_group = app_commands.Group(name="admin", description="Administration commands")
        self.asset_group = app_commands.Group(name="asset", description="Asset management commands")
        
        # Register groups
        self.tree.add_command(self.mod_group)
        self.tree.add_command(self.admin_group)
        self.tree.add_command(self.asset_group)
    
    async def setup_hook(self):
        """Setup hook - called before bot starts"""
        logger.info("Initializing Kynx SS 10...")
        
        # Start background tasks
        self.status_updater.start()
        self.cache_cleaner.start()
        
        # Sync commands
        await self.tree.sync()
        logger.info("Commands synced")
    
    async def close(self):
        """Cleanup on shutdown"""
        await self.api.close()
        await self.roblox.close()
        await super().close()
    
    def is_allowed(self, interaction: discord.Interaction) -> bool:
        """Check if user has allowed role"""
        if Config.ALLOWED_ROLE_ID == 0:
            return True
        if not isinstance(interaction.user, discord.Member):
            return False
        return any(r.id == Config.ALLOWED_ROLE_ID for r in interaction.user.roles)
    
    def is_admin(self, interaction: discord.Interaction) -> bool:
        """Check if user has admin role"""
        if Config.ADMIN_ROLE_ID == 0:
            return True
        if not isinstance(interaction.user, discord.Member):
            return False
        return any(r.id == Config.ADMIN_ROLE_ID for r in interaction.user.roles)
    
    @tasks.loop(seconds=60)
    async def status_updater(self):
        """Update bot status periodically"""
        try:
            stats = await self.api.get_stats()
            await self.change_presence(
                activity=discord.Activity(
                    type=discord.ActivityType.watching,
                    name=f"{stats.players_online} players • SS10"
                )
            )
        except:
            pass
    
    @tasks.loop(hours=1)
    async def cache_cleaner(self):
        """Clean old cache files"""
        try:
            # Clean exports older than 7 days
            cutoff = time.time() - (7 * 24 * 3600)
            for file in Config.EXPORTS_DIR.glob("*"):
                if file.stat().st_mtime < cutoff:
                    file.unlink()
                    logger.debug(f"Removed old export: {file.name}")
        except Exception as e:
            logger.error(f"Cache cleaner error: {e}")

# ==================== DECORATORS ====================
def permission_check(allowed_only: bool = True, admin_only: bool = False):
    """Permission check decorator"""
    def decorator(func):
        async def wrapper(interaction: discord.Interaction, *args, **kwargs):
            bot: KynxBot = interaction.client
            
            if admin_only and not bot.is_admin(interaction):
                await interaction.response.send_message(
                    embed=EmbedFactory.error("This command requires admin privileges!", interaction.user),
                    ephemeral=True
                )
                return
            
            if allowed_only and not bot.is_allowed(interaction):
                await interaction.response.send_message(
                    embed=EmbedFactory.error("You don't have permission to use this command!", interaction.user),
                    ephemeral=True
                )
                return
            
            return await func(interaction, *args, **kwargs)
        return wrapper
    return decorator

def handle_errors(func):
    """Error handling decorator"""
    async def wrapper(interaction: discord.Interaction, *args, **kwargs):
        try:
            return await func(interaction, *args, **kwargs)
        except Exception as e:
            logger.error(f"Command error in {func.__name__}: {e}", exc_info=True)
            
            if not interaction.response.is_done():
                await interaction.response.send_message(
                    embed=EmbedFactory.error(f"An error occurred: {str(e)}", interaction.user),
                    ephemeral=True
                )
            else:
                await interaction.followup.send(
                    embed=EmbedFactory.error(f"An error occurred: {str(e)}", interaction.user),
                    ephemeral=True
                )
    return wrapper

bot = KynxBot()

@bot.mod_group.command(name="kick", description="Kick a player from the server")
@app_commands.describe(user_id="Roblox User ID", reason="Reason for kick")
@permission_check(allowed_only=True)
@handle_errors
async def mod_kick(interaction: discord.Interaction, user_id: str, reason: str):
    await interaction.response.defer()
    
    # Validate user ID
    if not user_id.isdigit():
        await interaction.followup.send(
            embed=EmbedFactory.error("User ID must contain only numbers!", interaction.user)
        )
        return
    
    # Get Roblox user

    if notes:

def main():
    """Main entry point"""
    if not Config.TOKEN:
        logger.error("❌ No Discord token found in .env file!")
        return
    
    logger.info("🚀 Starting Kynx SS 10...")
    logger.info(f"📁 Data directory: {Config.BASE_DIR.absolute()}")
    
    try:
        bot.run(Config.TOKEN, log_handler=None)
    except KeyboardInterrupt:
        logger.info("👋 Shutting down...")
    except Exception as e:
        logger.error(f"💥 Fatal error: {e}", exc_info=True)

if __name__ == "__main__":
    main()
